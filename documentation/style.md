### [Space before blocks](http://eslint.org/docs/rules/space-before-blocks)

> require or disallow space before blocks


&#10003; Enabled (error)

	b();
}

	b();
}

	constructor(){}
}
*/

// Good
if (a) {
	b();
}

function a() {}

for (;;) {
	b();
}

try {} catch (a) {}

	constructor() {}
}

```

### [Space before function paren](http://eslint.org/docs/rules/space-before-function-paren)

&#10003; Enabled (error)

	// ...

	constructor () {
		// ...
	}

	bar () {
		// ...
	}
};
*/

// Good

	// ...
}

var bar = function () {
	// ...
};

	constructor() {
		// ...
	}
}

	bar() {
		// ...
	},
};

```
